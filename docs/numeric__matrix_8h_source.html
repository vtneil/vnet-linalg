479 T det() const {
480 static_assert(static_is_a_square_matrix(), "Can only find determinant of a square matrix.");
481 numeric_matrix_static_lu_t<T, Order> lu = vt::move(LU());
482 T det = 1;
483 size_t r_swaps = 0;
484 for (size_t i = 0; i < Order; ++i) {
485 det *= lu.u()[i][i];
486 if (lu.u()[i][i] == 0) return 0;
487 if (lu.l()[i][i] != 1) r_swaps++;

531 numeric_matrix_static_lu_t<T, Order> LU() const {
532 static_assert(static_is_a_square_matrix(), "Can only find LU decomposition of a square matrix.");
533 numeric_matrix_static_t<T, Order, Order> lower;
534 numeric_matrix_static_t<T, Order, Order> upper;
535 for (size_t i = 0; i < Order; ++i) {
536 for (size_t k = 0; k < Order; ++k) {
537 T sum_ = 0;
538 for (size_t j = 0; j < i; ++j) sum_ += lower[i][j] * upper[j][k];
539 upper[i][k] = vector_[i][k] - sum_;
540 }
541 for (size_t k = i; k < Order; ++k) {
542 if (i == k) lower[i][i] = 1;
543 else {
544 T sum_ = 0;
545 for (size_t j = 0; j < i; ++j) sum_ += lower[k][j] * upper[j][i];
546 lower[k][i] = (vector_[k][i] - sum_) / upper[i][i];
547 }
548 }
549 }
550 return {lower, upper};
551 }

559 numeric_matrix_static_t RRE() const {
560 numeric_matrix_static_t m(*this);
561 size_t lead = 0;
562 for (size_t r = 0; r < Row; ++r) {
563 if (lead >= Col) return m;
564 size_t i;
565 for (i = r; m[i][lead] == 0;) {
566 ++i;
567 if (i == Row) {
568 i = r;
569 ++lead;
570 if (lead == Col) return m;
571 }
572 }
573 m[i].swap(m[r]);
574 T val = m[r][lead];
575 for (size_t j = 0; j < Col; ++j) m[r][j] /= val;
576 for (i = 0; i < Row; ++i) {
577 if (i != r) {
578 val = m[i][lead];
579 for (size_t j = 0; j < Col; ++j)
580 m[i][j] -= val * m[r][j];
581 }
582 }
583 ++lead;
584 }
585 m.fix_zero();
586 return m;
587 }

855 static numeric_matrix_static_t<T, OSize, OSize> inv_lt(numeric_matrix_static_t<T, OSize, OSize> &L) {
856 static_assert(static_is_a_square_matrix(), "Can only inverse a square matrix.");
857 numeric_matrix_static_t<T, OSize, OSize> X;
858 for (size_t k = 0; k < OSize; ++k) {
859 X[k][k] = 1 / L[k][k];
860 for (size_t i = k + 1; i < OSize; ++i) {
861 T acc = 0;
862 for (size_t j = k; j < i; ++j)
863 acc -= L[i][j] * X[j][k];
864 X[i][k] = acc / L[i][i];
865 }
866 }
867 L = vt::move(X);
868 return L;
869 }
