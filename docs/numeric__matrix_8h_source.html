329 numeric_matrix_static_t<T, Order, Order> operator^(size_t n) {
330 static_assert(static_is_a_square_matrix(), "Non-square matrix can\'t use power operator.");
331 if (n == 0) return identity();
332 if (n == 1) return numeric_matrix_static_t(*this);
333 numeric_matrix_static_t<T, Order, Order> base(*this);
334 numeric_matrix_static_t<T, Order, Order> product(vt::move(identity()));
335 while (n > 0) {
336 if (n % 2 == 1) product.operator*=(base);
337 if (n > 1) base *= base;
338 n >>= 1;
339 }
340 return product;

498 T det = 1;
499 size_t r_swaps = 0;
500 for (size_t i = 0; i < Order; ++i) {
501 det *= lu.u()[i][i];
502 if (lu.u()[i][i] == 0) return 0;

505 return (r_swaps % 2 == 0) ? det : -det;

552 numeric_matrix_static_lu_t<T, Order> LU() const {
553 static_assert(static_is_a_square_matrix(), "Can only find LU decomposition of a square matrix.");
554 numeric_matrix_static_t<T, Order, Order> lower;
555 numeric_matrix_static_t<T, Order, Order> upper;
556 for (size_t i = 0; i < Order; ++i) {
557 for (size_t k = 0; k < Order; ++k) {
558 T sum_ = 0;
559 for (size_t j = 0; j < i; ++j) sum_ += lower[i][j] * upper[j][k];
560 upper[i][k] = vector_[i][k] - sum_;
561 }
562 for (size_t k = i; k < Order; ++k) {
563 if (i == k) lower[i][i] = 1;
564 else {
565 T sum_ = 0;
566 for (size_t j = 0; j < i; ++j) sum_ += lower[k][j] * upper[j][i];
567 lower[k][i] = (vector_[k][i] - sum_) / upper[i][i];
568 }
569 }
570 }
571 return {lower, upper};
572 }

580 numeric_matrix_static_t RRE() const {
581 numeric_matrix_static_t m(*this);
582 size_t lead = 0;
583 for (size_t r = 0; r < Row; ++r) {
584 if (lead >= Col) return m;
585 size_t i;
586 for (i = r; m[i][lead] == 0;) {
587 ++i;
588 if (i == Row) {
589 i = r;
590 ++lead;
591 if (lead == Col) return m;
592 }
593 }
594 m[i].swap(m[r]);
595 T val = m[r][lead];
596 for (size_t j = 0; j < Col; ++j) m[r][j] /= val;
597 for (i = 0; i < Row; ++i) {
598 if (i != r) {
599 val = m[i][lead];
600 for (size_t j = 0; j < Col; ++j)
601 m[i][j] -= val * m[r][j];
602 }
603 }
604 ++lead;
605 }
606 m.fix_zero();
607 return m;
608 }

881 template<size_t OSize>
882 static numeric_matrix_static_t<T, OSize, OSize> inv_lt(numeric_matrix_static_t<T, OSize, OSize> &L) {
883 static_assert(static_is_a_square_matrix(), "Can only inverse a square matrix.");
884 numeric_matrix_static_t<T, OSize, OSize> X;
885 for (size_t k = 0; k < OSize; ++k) {
886 X[k][k] = 1 / L[k][k];
887 for (size_t i = k + 1; i < OSize; ++i) {
888 T acc = 0;
889 for (size_t j = k; j < i; ++j)
890 acc -= L[i][j] * X[j][k];
891 X[i][k] = acc / L[i][i];
892 }
893 }
894 L = vt::move(X);
895 return L;
896 }
