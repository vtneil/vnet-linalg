291 static_assert(static_is_a_square_matrix(), "Non-square matrix can\'t use power operator.");
292 if (n == 0) return identity();

504 numeric_matrix_static_lu_t<T, Order> LU() const {
505 static_assert(static_is_a_square_matrix(), "Can only find LU decomposition of a square matrix.");
506 numeric_matrix_static_t<T, Order, Order> lower;
507 numeric_matrix_static_t<T, Order, Order> upper;
508 for (size_t i = 0; i < Order; ++i) {
509 for (size_t k = 0; k < Order; ++k) {
510 T sum_ = 0;
511 for (size_t j = 0; j < i; ++j) sum_ += lower[i][j] * upper[j][k];
512 upper[i][k] = vector_[i][k] - sum_;
513 }
514 for (size_t k = i; k < Order; ++k) {
515 if (i == k) lower[i][i] = 1;
516 else {
517 T sum_ = 0;
518 for (size_t j = 0; j < i; ++j) sum_ += lower[k][j] * upper[j][i];
519 lower[k][i] = (vector_[k][i] - sum_) / upper[i][i];
520 }
521 }
522 }
523 return {lower, upper};
524 }

532 numeric_matrix_static_t RRE() const {
533 numeric_matrix_static_t m(*this);
534 size_t lead = 0;
535 for (size_t r = 0; r < Row; ++r) {
536 if (lead >= Col) return m;
537 size_t i;
538 for (i = r; m[i][lead] == 0;) {
539 ++i;
540 if (i == Row) {
541 i = r;
542 ++lead;
543 if (lead == Col) return m;
544 }
545 }
546 m[i].swap(m[r]);
547 T val = m[r][lead];
548 for (size_t j = 0; j < Col; ++j) m[r][j] /= val;
549 for (i = 0; i < Row; ++i) {
550 if (i != r) {
551 val = m[i][lead];
552 for (size_t j = 0; j < Col; ++j)
553 m[i][j] -= val * m[r][j];
554 }
555 }
556 ++lead;
557 }
558 m.fix_zero();
559 return m;
560 }

805 template<size_t OSize>
806 static numeric_matrix_static_t<T, OSize, OSize> inv_lt(numeric_matrix_static_t<T, OSize, OSize> &L) {
807 static_assert(static_is_a_square_matrix(), "Can only inverse a square matrix.");
808 numeric_matrix_static_t<T, OSize, OSize> X;
809 for (size_t k = 0; k < OSize; ++k) {
810 X[k][k] = 1 / L[k][k];
811 for (size_t i = k + 1; i < OSize; ++i) {
812 T acc = 0;
813 for (size_t j = k; j < i; ++j)
814 acc -= L[i][j] * X[j][k];
815 X[i][k] = acc / L[i][i];
816 }
817 }
818 L = vt::move(X);
819 return L;
820 }
