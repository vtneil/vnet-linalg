296 numeric_matrix_static_t<T, Order, Order> operator^(size_t n) {
297 static_assert(static_is_a_square_matrix(), "Non-square matrix can\'t use power operator.");
298 if (n == 0) return identity();
299 if (n == 1) return numeric_matrix_static_t(*this);
300 numeric_matrix_static_t<T, Order, Order> base(*this);
301 numeric_matrix_static_t<T, Order, Order> product(vt::move(identity()));
302 while (n > 0) {
303 if (n % 2 == 1) product.operator*=(base);
304 if (n > 1) base *= base;
305 n >>= 1;
306 }
307 return product;
308 }

458 T det() const {
459 static_assert(static_is_a_square_matrix(), "Can only find determinant of a square matrix.");
460 numeric_matrix_static_lu_t<T, Order> lu = vt::move(LU());
461 T det = 1;
462 size_t r_swaps = 0;
463 for (size_t i = 0; i < Order; ++i) {
464 det *= lu.u()[i][i];
465 if (lu.u()[i][i] == 0) return 0;
466 if (lu.l()[i][i] != 1) r_swaps++;
467 }
468 return (r_swaps % 2 == 0) ? det : -det;

510 numeric_matrix_static_lu_t<T, Order> LU() const {
511 static_assert(static_is_a_square_matrix(), "Can only find LU decomposition of a square matrix.");
512 numeric_matrix_static_t<T, Order, Order> lower;
513 numeric_matrix_static_t<T, Order, Order> upper;
514 for (size_t i = 0; i < Order; ++i) {
515 for (size_t k = 0; k < Order; ++k) {
516 T sum_ = 0;
517 for (size_t j = 0; j < i; ++j) sum_ += lower[i][j] * upper[j][k];
518 upper[i][k] = vector_[i][k] - sum_;
519 }
520 for (size_t k = i; k < Order; ++k) {
521 if (i == k) lower[i][i] = 1;
522 else {
523 T sum_ = 0;
524 for (size_t j = 0; j < i; ++j) sum_ += lower[k][j] * upper[j][i];
525 lower[k][i] = (vector_[k][i] - sum_) / upper[i][i];
526 }
527 }
528 }
529 return {lower, upper};
530 }

538 numeric_matrix_static_t RRE() const {
539 numeric_matrix_static_t m(*this);
540 size_t lead = 0;
541 for (size_t r = 0; r < Row; ++r) {
542 if (lead >= Col) return m;
543 size_t i;
544 for (i = r; m[i][lead] == 0;) {
545 ++i;
546 if (i == Row) {
547 i = r;
548 ++lead;
549 if (lead == Col) return m;
550 }
551 }
552 m[i].swap(m[r]);
553 T val = m[r][lead];
554 for (size_t j = 0; j < Col; ++j) m[r][j] /= val;
555 for (i = 0; i < Row; ++i) {
556 if (i != r) {
557 val = m[i][lead];
558 for (size_t j = 0; j < Col; ++j)
559 m[i][j] -= val * m[r][j];
560 }
561 }
562 ++lead;
563 }
564 m.fix_zero();
565 return m;
566 }

811 template<size_t OSize>
812 static numeric_matrix_static_t<T, OSize, OSize> inv_lt(numeric_matrix_static_t<T, OSize, OSize> &L) {
813 static_assert(static_is_a_square_matrix(), "Can only inverse a square matrix.");
814 numeric_matrix_static_t<T, OSize, OSize> X;
815 for (size_t k = 0; k < OSize; ++k) {
816 X[k][k] = 1 / L[k][k];
817 for (size_t i = k + 1; i < OSize; ++i) {
818 T acc = 0;
819 for (size_t j = k; j < i; ++j)
820 acc -= L[i][j] * X[j][k];
821 X[i][k] = acc / L[i][i];
822 }
823 }
824 L = vt::move(X);
825 return L;
826 }
