518 numeric_matrix_static_lu_t<T, Order> LU() const {
519 static_assert(static_is_a_square_matrix(), "Can only find LU decomposition of a square matrix.");
520 numeric_matrix_static_t<T, Order, Order> lower;
521 numeric_matrix_static_t<T, Order, Order> upper;
522 for (size_t i = 0; i < Order; ++i) {
523 for (size_t k = 0; k < Order; ++k) {
524 T sum_ = 0;
525 for (size_t j = 0; j < i; ++j) sum_ += lower[i][j] * upper[j][k];
526 upper[i][k] = vector_[i][k] - sum_;
527 }
528 for (size_t k = i; k < Order; ++k) {
529 if (i == k) lower[i][i] = 1;
530 else {
531 T sum_ = 0;
532 for (size_t j = 0; j < i; ++j) sum_ += lower[k][j] * upper[j][i];
533 lower[k][i] = (vector_[k][i] - sum_) / upper[i][i];
534 }
535 }
536 }
537 return {lower, upper};
538 }

546 numeric_matrix_static_t RRE() const {
547 numeric_matrix_static_t m(*this);
548 size_t lead = 0;
549 for (size_t r = 0; r < Row; ++r) {
550 if (lead >= Col) return m;
551 size_t i;
552 for (i = r; m[i][lead] == 0;) {
553 ++i;
554 if (i == Row) {
555 i = r;
556 ++lead;
557 if (lead == Col) return m;
558 }
559 }
560 m[i].swap(m[r]);
561 T val = m[r][lead];
562 for (size_t j = 0; j < Col; ++j) m[r][j] /= val;
563 for (i = 0; i < Row; ++i) {
564 if (i != r) {
565 val = m[i][lead];
566 for (size_t j = 0; j < Col; ++j)
567 m[i][j] -= val * m[r][j];
568 }
569 }
570 ++lead;
571 }
572 m.fix_zero();
573 return m;
574 }

842 static numeric_matrix_static_t<T, OSize, OSize> inv_lt(numeric_matrix_static_t<T, OSize, OSize> &L) {
843 static_assert(static_is_a_square_matrix(), "Can only inverse a square matrix.");
844 numeric_matrix_static_t<T, OSize, OSize> X;
845 for (size_t k = 0; k < OSize; ++k) {
846 X[k][k] = 1 / L[k][k];
847 for (size_t i = k + 1; i < OSize; ++i) {
848 T acc = 0;
849 for (size_t j = k; j < i; ++j)
850 acc -= L[i][j] * X[j][k];
851 X[i][k] = acc / L[i][i];
852 }
853 }
854 L = vt::move(X);
855 return L;
